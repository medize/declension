        <p id="question">Loading...</p>

        <input type="text" id="answer" placeholder="Type your answer here">

        <button onclick="checkAnswer()">Submit</button>

        <p id="feedback"></p>

        <script>

            const nouns = [

                { base: 'λόγος', genitive: 'λόγου', declension: '2nd' },

                { base: 'γυνή', genitive: 'γυναικός', declension: '3rd' },

                { base: 'ὥρα', genitive: 'ὥρας', declension: '1st' },

            ];

            let currentNoun = null;

            let currentCase = 'nominative';

            let currentNumber = 'singular';

            const cases = ['nominative', 'genitive', 'dative', 'accusative', 'vocative'];

            function getRandomNoun() {

                const noun = nouns[Math.floor(Math.random() * nouns.length)];

                currentNoun = noun;

                currentCase = cases[Math.floor(Math.random() * cases.length)];

                currentNumber = Math.random() < 0.5 ? 'singular' : 'plural';

                document.getElementById('question').textContent = `Decline ${noun.base} (${noun.declension} declension) in the ${currentCase} ${currentNumber}.`;

            }

            function checkAnswer() {

                const userAnswer = document.getElementById('answer').value.trim();

                const correctAnswer = getCorrectForm();

                const feedback = document.getElementById('feedback');

                if (userAnswer === correctAnswer) {

                    feedback.textContent = 'Correct!';

                    feedback.style.color = 'green';

                } else {

                    feedback.textContent = `Incorrect. The correct form is ${correctAnswer}.`;

                    feedback.style.color = 'red';

                }

                getRandomNoun();

                document.getElementById('answer').value = '';

            }

            function getCorrectForm() {

                if (currentCase === 'nominative' && currentNumber === 'singular') {

                    return currentNoun.base;

                } else if (currentCase === 'genitive' && currentNumber === 'singular') {

                    return currentNoun.genitive;

                }

                return 'TBD'; // Add logic for other cases.

            }

            getRandomNoun();

        </script>
